add_namespace = super

country_event = {	#ww2
	id = super.1
	title = super.1.t
	desc = super.1.d
	picture = GFX_report_event_diplomatic_message

	is_triggered_only = no
	
	fire_only_once = yes
	
	hidden = yes
	
	trigger = {
		AND = {
			GER = { has_war_with = FRA }
			NOT = {
				has_global_flag = Super_2nd_weltkrieg
			}
		}
	}

	option = {
		name = super.1.a
		
       		set_global_flag = zweite_weltkrieg
		set_global_flag = Super_Event_Visible
		set_global_flag = second_world_war
		play_song = "super_second_world_war"
		
		set_global_flag = Super_2nd_weltkrieg
	}
}
